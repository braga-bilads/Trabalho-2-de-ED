116void CodificaArq(FILE *arq, Arv *Huffman, VetChar *Vetor, FILE *arqSaida)
117{
118 // 14MB
119 bitmap *saida = bitmapInit(112000000);
120 int tam = QntdFolhas(Huffman);
121
122 Tabela *tab = MontandoTabela(Huffman, Vetor, tam);
123
124 // Escreve o tamanho total do arquivo - Subcabecalho do arquivo
125 unsigned long int TAM_TOTAL = CalculaTamTotal(Vetor, tab, tam);
126 fprintf(arqSaida, "%ld", TAM_TOTAL);
127
128 // Codificacao do arquivo de fato
129 unsigned char aux;
130 int index;
131 bitmap *codificando;
132 while (!feof(arq))
133 {
134 // Ao ler um byte
135 if (fread(&aux, 1, 1, arq))
136 {
137 // Procura o mesmo na tabela
138 index = 0;
139 while (index < tam && tab->carac[index] != aux)
140 {
141 index++;
142 }
143 codificando = tab->codigo[index];
144
145 // Escreve no bitmap de saida o codigo do mesmo
146 for (int i = 0; i < bitmapGetLength(codificando); i++)
147 {
148 bitmapAppendLeastSignificantBit(saida,
149 bitmapGetBit(codificando, i));
150 }
151 }
152
153 // Escreve em disco caso buffer esteja maior que 7MB (50%) e tenha bytes completos
154 if (bitmapGetLength(saida) > 56000000 && (bitmapGetLength(saida) % 8) == 0)
155 {
156 fwrite(bitmapGetContents(saida), 1, bitmapGetLength(saida) / 8, arqSaida);
157 bitmapLibera(saida);
158 bitmap *saida = bitmapInit(112000000);
159 }
160 }
161
162 // Garantia de que caso tenha algo a ser escrito, seja escrito
163 if (bitmapGetLength(saida))
164 {
165 // completa o resto do byte com 0's
166 CompletaByteBitmap(saida);
167 fwrite(bitmapGetContents(saida), 1, bitmapGetLength(saida) / 8, arqSaida);
168 }
169
170 bitmapLibera(saida);
171 LiberaTabela(tab, tam);
172 rewind(arq);
173}

175Tabela *MontandoTabela(Arv *Huffman, VetChar *vetor, int tam)
176{
177 // Montando tabela de codificacao
178
179 Tabela *tab = (Tabela *)malloc(sizeof(Tabela));
180
181 int count = 0;
182
183 tab->carac = (unsigned char *)malloc(tam * sizeof(unsigned char));
184 tab->codigo = (bitmap **)malloc(tam * sizeof(bitmap *));
185
186 // Preenchendo a tabela
187 for (int i = 0; i < MAX_VET; i++)
188 {
189 if (VetGetPos(vetor, i) != 0)
190 {
191 // para cada caracter existente no arquivo de saida um código é criado aqui
192 // o i representa o caracter no vetor de frequencia
193 tab->carac[count] = (unsigned char)i;
194 tab->codigo[count] = CodificaChar(Huffman, tab->carac[count]);
195 count++;
196 }
197 }
198 return tab;
199}

212unsigned long int CalculaTamTotal(VetChar *Vetor, Tabela *tab, int tam)
213{
214 unsigned long int TAM_TOTAL = 0;
215 for (int i = 0; i < MAX_VET; i++)
216 {
217 if (VetGetPos(Vetor, i) != 0)
218 {
219 for (int j = 0; j < tam; j++)
220 {
221 if (tab->carac[j] == (unsigned char)i)
222 {
223 TAM_TOTAL += VetGetPos(Vetor, i) * bitmapGetLength(tab->codigo[j]);
224 }
225 }
226 }
227 }
228 return TAM_TOTAL;
