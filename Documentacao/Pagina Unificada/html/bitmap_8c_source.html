71bitmap *bitmapInit(unsigned int max_size)
72{
73 bitmap *bm;
74 bm = (bitmap *)malloc(sizeof(bitmap));
75 // definir tamanho maximo em bytes, com arredondamento para cima
76 unsigned int max_sizeInBytes = (max_size + 7) / 8;
77 // alocar espaco de memoria para o tamanho maximo em bytes
78 bm->contents = calloc(max_sizeInBytes, sizeof(char));
79 // verificar alocacao de memoria
80 assert(bm->contents != NULL, "Erro de alocacao de memoria.");
81 // definir valores iniciais para tamanho maximo e tamanho atual
82 bm->max_size = max_size;
83 bm->length = 0;
84 return bm;
85}

94unsigned char bitmapGetBit(bitmap *bm, unsigned int index) // index in bits
95{
96 // verificar se index<bm.length, pois caso contrario, index e' invalido
97 assert(index < bm->length, "Acesso a posicao inexistente no mapa de bits.");
98 // index/8 e' o indice do byte que contem o bit em questao
99 // 7-(index%8) e' o deslocamento do bit em questao no byte
100 return (bm->contents[index / 8] >> (7 - (index % 8))) & 0x01;
101}

110static void bitmapSetBit(bitmap *bm, unsigned int index, unsigned char bit)
111{
112 // verificar se index<bm->length, pois caso contrario, index e' invalido
113 assert(index < bm->length, "Acesso a posicao inexistente no mapa de bits.");
114 // index/8 e' o indice do byte que contem o bit em questao
115 // 7-(index%8) e' o deslocamento do bit em questao no byte
116 bit = bit & 0x01;
117 bit = bit << (7 - (index % 8));
118 bm->contents[index / 8] = bm->contents[index / 8] | bit;
119}

129void bitmapAppendLeastSignificantBit(bitmap *bm, unsigned char bit)
130{
131 // verificar se bm->length<bm->max_size, caso contrario, o bitmap esta' cheio
132 assert(bm->length < bm->max_size, "Tamanho maximo excedido no mapa de bits.");
133 // como um bit sera' adicionado, devemos incrementar o tamanho do mapa de bits
134 bm->length++;
135 bitmapSetBit(bm, bm->length - 1, bit);
136}

149void EscreveChar(bitmap *mapa, unsigned char letrona)
150{
151 unsigned int aux = (unsigned int)letrona;
152 unsigned int enviado = 0;
153
154 bitmap *temporarioInvertido = bitmapInit(8);
155 for (int i = 0; i < 8; i++)
156 {
157 bitmapAppendLeastSignificantBit(temporarioInvertido, aux % 2);
158 aux = aux / 2;
159 }
160 for (int i = 0; i < 8; i++)
161 {
162 bitmapAppendLeastSignificantBit(mapa, bitmapGetBit(temporarioInvertido, 7 - i));
163 enviado = enviado / 10;
164 }
165 bitmapLibera(temporarioInvertido);
166}
